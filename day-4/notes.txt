* queries

create table user (id integer primary  key auto_increment, firstname varchar(100), lastname varchar(100), address varchar(500), email varchar(100), password varchar(150), phone varchar(20), profileImage varchar(100));


create table blog (id integer primary key auto_increment, title varchar(200), contents varchar(5000), image varchar(100), userId integer);

create table comment (id integer primary key auto_increment, userId integer, blogId integer, comment varchar(5000));

create table rating (id integer primary key auto_increment, userId integer, blogId integer, rating float);
